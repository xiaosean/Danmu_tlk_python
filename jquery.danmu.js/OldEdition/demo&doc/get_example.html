用于查询弹幕的后端php网页示例:

<?php header('Content-type:text/html;charset=utf8');
$conn = @ mysql_connect("localhost", "root", "") or die("datebase can`t been connected"); //链接数据库，替换成自己的
mysql_select_db("danmu", $conn); //选择弹幕列表所存放数据库，替换成自己的
mysql_query("set names 'utf8'"); // 设置编码
$sql="SELECT `danmu` FROM `danmu`";  //在这里弹幕所存放的数据表名称为danmu，该表只有一个text型字段，字段名名也为danmu
$query=mysql_query($sql); //
echo $danmu; echo "[";   //数组的前中括号
$first=0; 
while($row=mysql_fetch_array($query)){ 
if ($first) { 
echo ",";        //数组元素分隔符逗号，条件判断，数组前一个元素前不用逗号
} 
$first=1; 
echo "'".$row['danmu']."'";  //danmu对象（数据库里该字段每条记录都是一个text形式的弹幕对象）
} 
echo "]";  //数组的后中括号

?>
